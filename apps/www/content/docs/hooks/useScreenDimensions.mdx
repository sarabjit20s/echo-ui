## Usage

```tsx title="MyComponent.tsx"
import React from 'react';
import { View } from 'react-native';

import { useScreenDimensions } from '@/hooks/useScreenDimensions';

export function MyComponent() {
    const { width, height } = useScreenDimensions();

    return (
        <View
            style={{
                width,
                height,
                backgroundColor: 'red',
            }}
        >
            {/* ... */}
        </View>
    )
}
```
