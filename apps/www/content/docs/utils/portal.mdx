## Setup

<Steps>
<Step>
Wrap your app with `PortalProvider` component.
<Tabs groupId='framework' persist items={['React Native', 'Expo']}>
```tsx tab="React Native" title="App.tsx"
import { PortalProvider } from '@/utils/portal';

export default function App() {
  return (
    <PortalProvider>
      {/* Your app */}
    </PortalProvider>
  )
}
```

```tsx tab="Expo" title="app/_layout.tsx"
import { PortalProvider } from '@/utils/portal';

export default function RootLayout() {
  return (
    <PortalProvider>
      {/* Your app */}
    </PortalProvider>
  )
}
```

## Usage

```tsx title="MyComponent.tsx"
import React from 'react';
import { View } from 'react-native';

import { Portal } from '@/utils/portal';

export function MyComponent() {

    return (
        <View>
            <Portal>
                {/* Your portal content */}
            </Portal>
        </View>
    )
}
```
