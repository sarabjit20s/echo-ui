## Usage

```tsx title="MyAvatar.tsx"
import {
  Avatar,
  AvatarFallback,
  AvatarImage,
  AvatarProps,
  AvatarText,
} from '@/components/ui/Avatar';
import { Text } from '@/components/ui/Text';

export function MyAvatar() {
  return (
    <Avatar>
      <AvatarImage src="https://avatar.iran.liara.run/public?username=John+Doe" />
      <AvatarFallback>
        <AvatarText>
          JD
        </AvatarText>
      </AvatarFallback>
    </Avatar>
  );
}
```
